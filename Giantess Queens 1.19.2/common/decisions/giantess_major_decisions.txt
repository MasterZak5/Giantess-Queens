### Form the Titanes ###
form_titanes_decision = {
	picture = {
		reference = "gfx/interface/illustrations/decisions/decision_misc.dds" # change to correct image
	}
	desc = form_titanes_decision_desc
	selection_tooltip = form_titanes_decision_tooltip
	decision_group_type = major

	ai_check_interval_by_tier = {
		barony = 0
		county = 0
		duchy = 12
		kingdom = 6
		empire = 0
		hegemony = 0
	}

	is_shown = {
		is_ruler = yes
		is_playable_character = yes
		has_trait = giantess
        culture = { has_cultural_pillar = heritage_byzantine }
		OR = {
			government_has_flag = government_is_feudal
			government_has_flag = government_is_republic
			government_has_flag = government_is_clan
			government_allows = administrative
		}
        NOT = {  #Can only do it once.
			is_target_in_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:flag_formed_titanes_kingdom
			}
		}
	}

	is_valid = {
		prestige_level >= 4
		top_liege = this
        culture = { has_cultural_pillar = heritage_byzantine }
		has_trait = giantess

		completely_controls = title:d_achaia
		completely_controls = title:d_aegean_islands
		completely_controls = title:d_athens
		completely_controls = title:d_krete
		completely_controls = title:d_thessalonika
		completely_controls = title:d_thessaly
	}

	is_valid_showing_failures_only = {
		is_landed = yes
		is_available_adult = yes
		is_imprisoned = no
        is_at_war = no
	}

	effect = {
		custom_tooltip = form_titanes_decision_effect_tooltip
		show_as_tooltip = {
			form_titanes_scripted_effect = yes
		}
		trigger_event = titanes_formation.0001
        hidden_effect = {
			add_to_global_variable_list = {
				name = unavailable_unique_decisions
				target = flag:flag_formed_titanes_kingdom
			}
			set_global_variable = {
				name = flag_formed_titanes_kingdom
				value = root
			}
		}
	}
	
	ai_potential = {
		always = yes
	}

	ai_will_do = {
		base = 100
	}
}
