create_titanes_scripted_effect = {
	primary_title = { save_scope_as = old_primary_title }
	if = {
		limit = { has_dlc_feature = legends }
		create_legend_seed = {
			type = legitimizing
			quality = illustrious
			chronicle = new_title
			properties = {
				title = title:k_titanes
				founder = root
			}
		}
	}
	#Create Titanes, shift dejure/history/laws.

	if = {
		limit = { capital_county = { this = title:c_thessalia } }
		title:k_titanes  = {
			set_capital_county = title:c_thessalia
		}
	}

	hidden_effect = {
		every_held_title = { #Should shift all dejure of all Duchies owned at the time.
			title_tier = duchy

			set_de_jure_liege_title = title:k_titanes
		}
		every_duchy = {
			limit = {
				NOT = { exists = hegemony }
				any_de_jure_county = {
					percent >= 0.6
					holder = {
						OR = {
							this = root
							any_liege_or_above = { this = root }
						}
					}
				}
			}
			set_de_jure_liege_title = title:k_titanes
		}
	}
	create_title_and_vassal_change = {
		type = created
		save_scope_as = change
		add_claim_on_loss = no
	}
	title:k_titanes = {
		change_title_holder = {
			holder = root
			change = scope:change
		}
	}
	resolve_title_and_vassal_change = scope:change
	if = { # Move your Title MaAs to your new title if you have any
		limit = {
			government_allows = administrative
			primary_title = {
				any_title_maa_regiment = { }
			}
		}
		hidden_effect = {
			primary_title = { transfer_title_maa_ownership = title:k_titanes }
		}
	}
	if = { # If you have founded the Varangian Guard, you get to keep the benefits
		limit = {
			scope:old_primary_title = { has_variable = founded_varangian_guard }
		}
		hidden_effect = {
			title:k_titanes = { set_variable = founded_varangian_guard }
		}
	}
	hidden_effect = { set_primary_title_to = title:k_titanes }
	
	if = { #Automatically move capital to Thessalia.
		limit = {
			NOT = { capital_county = { this = title:c_thessalia } }
			culture = { has_cultural_pillar = heritage_byzantine }
		}
		hidden_effect = {
			if = { #Usurp if not held personally.
				limit = {
					NOT = { this = title:c_thessalia.holder }
				}
				create_title_and_vassal_change = {
					type = returned
					save_scope_as = change
					add_claim_on_loss = no
				}
				title:c_thessalia = {
					change_title_holder = {
						holder = root
						change = scope:change
					}
				}
				resolve_title_and_vassal_change = scope:change
			}
		}
		root = { set_realm_capital = title:c_thessalia }
	}
}
