doctrine_giantess_acceptance_demonized = {
    piety_cost = {
        value = faith_doctrine_cost_low
        if = {
            limit = { has_doctrine = doctrine_giantess_acceptance_demonized }
            multiply = faith_unchanged_doctrine_cost_mult
        }
        else_if = {
            limit = { has_doctrine = doctrine_giantess_acceptance_accepted }
            multiply = faith_changed_doctrine_cost_mult_two_step
         }
        else_if = {
            limit = { has_doctrine = doctrine_giantess_acceptance_worshipped }
            multiply = faith_changed_doctrine_cost_mult_four_step
        }
    }
    traits = {
        sins = { giantess }
    }
    parameters = {
        giantess_demonized = yes
    }
}
