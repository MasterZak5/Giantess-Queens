#Success in temple, you win
giantess_religion.0009 = {
	type = character_event
	title = giantess_religion.0009.t
	desc = {
		first_valid = {
			triggered_desc = {
			    trigger = { has_character_flag =  mouth_struggle }
				desc = giantess_religion.0008.desc_struggle
			}
			triggered_desc = {
			    trigger = { has_character_flag =  mouth_worship }
				desc = giantess_religion.0008.desc_worship
			}
		}
		desc = giantess_religion.0009.desc
		first_valid = {
			triggered_desc = {
			    trigger = { is_female = yes }
				desc = giantess_religion.0009.desc_female
			}
			triggered_desc = {
			    trigger = { is_female = no }
				desc = giantess_religion.0009.desc_male
			}
		}
	}
	left_portrait = {
		character = root
	}
	right_portrait = {
		character = scope:giantess
	}
	theme = battle
	override_background = {
		reference = temple
	}
	immediate = {
		set_global_variable = {
			name = giantesses_founded
			value = yes
		}
		remove_character_flag = deciphered_map
	}
	#Become main giantess
	option = {
		name = giantess_religion.0009.desc.a
		trigger = { is_female = yes }
		make_giantess_effect = yes
		add_courtier = scope:giantess
		trigger_event = { id = giantess_religion.0010  days = 10 }
		every_independent_ruler = {
			limit = {
				highest_held_title_tier >= tier_duchy
				age >= 18
				is_ai = yes
				is_landed = yes
				NOT = { has_trait = giantess }
			}
			random = {
				chance = 50
				trigger_event = giantess_religion.00011
			}
		}
	}

	#Giantess takes throne
	option = {
		name = giantess_religion.0009.desc.b
		if = {
			limit = { is_female = no }
			every_spouse = { divorce = root }
			marry = scope:giantess
		}
		create_title_and_vassal_change = {
			type = swear_fealty
			save_scope_as = change
			add_claim_on_loss = no
		}
		root = {
			capital_county = {
				change_title_holder = {
					holder = scope:giantess
					change = scope:change
					take_baronies = no
				}
			}
			every_held_title = {
				if = {
					limit = {
						tier = root.primary_title.tier
					}
					change_title_holder = {
						holder = scope:giantess
						change = scope:change
					}
				}
				else_if = {
					limit = {
						OR = {
							NOT = { exists = scope:secondary_title }
							tier > scope:secondary_title.tier
						}
					}
					save_temporary_scope_as = secondary_title
				}
			}
		}
			
		root = {
			if = {
				limit = {
					exists = scope:secondary_title
				}
				every_vassal = {
					limit = {
						primary_title.tier >= scope:secondary_title.tier
						NOT = {
							primary_title.tier = tier_barony
						}
					}
					change_liege = {
						liege = scope:giantess
						change = scope:change
					}
				}
			}
		}
			
		root = {
			change_liege = {
				liege = scope:giantess
				change = scope:change
			}
		}
		resolve_title_and_vassal_change = scope:change
		scope:giantess = { trigger_event = { id = giantess_religion.0010  days = 10 } }
		
		every_independent_ruler = {
			limit = {
				highest_held_title_tier >= tier_duchy
				age >= 25
				NOT = { has_trait = giantess }
				is_ai = yes
				is_landed = yes
			}
			random = {
				chance = 50
				trigger_event = giantess_religion.00011
			}
		}
	}
}
