#Invite to feast
giantess_feast.0001 = {
	type = letter_event
	desc = giantess_feast.0001.desc
	sender = scope:sender
	opening = {
		desc = feast.1501.opening
	}
	trigger = {
		is_available_for_activity_trigger = yes
		exists = scope:activity.activity_host
		NOT = { exists = var:is_handling_feast_invitation }
	}
	immediate = {
		scope:activity.activity_host = {
			save_scope_as = sender
		}
		scope:activity.activity_location = {
			barony = {
				save_scope_as = barony
			}
		}
		set_variable = {
			name = is_handling_feast_invitation
			value = scope:sender
			days = 20
		}
	}
	#Accept
	option = {
		name = giantess_feast.0001.a
		play_music_cue = mx_cue_banquet
		accept_activity_invite = scope:activity
		set_variable = {
			name = booked_for_a_party
			value = scope:sender
			days = 20
		}
		stress_impact = {
			comfort_eater = minor_stress_loss
			drunkard = minor_stress_loss
		}
		ai_chance = {
			base = 100
			opinion_modifier = {
				opinion_target = scope:sender
			}
		}
	}

	#Decline
	option = {
		name = giantess_feast.0001.b
		decline_activity_invite = scope:activity
		stress_impact = {
			inappetetic = minor_stress_loss
			reclusive = minor_stress_loss
		}
		ai_chance = {
			base = 10
			ai_value_modifier = {
				ai_boldness = 0.2
			}
		}
	}
}

#Vassals needs to provide food
giantess_feast.1000 = {
	type = letter_event
	desc = giantess_feast.1000.desc
	sender = scope:giantess
	opening = {
		desc = feast.1501.opening
	}
	immediate = {
		save_scope_as = vassal
	}
	#Send peasants
	option = {
		name = giantess_feast.1000.a
		giantess_send_peasants_feast_effect = yes
		tiny_loyalty_change = { CHANGE = minor_loyalty_gain }
		if = {
			limit = {
				faith = { has_doctrine_parameter = giantess_worshipped }
			}
			add_piety = medium_piety_gain
		}
		add_character_modifier = {
			modifier = giantess_feast_vassal_sent_peasants
			years = 5
		}
		scope:giantess = {
			send_interface_message = {
				left_icon = scope:vassal
				type = event_prowess_good_desc
				title = giantess_feast.1000.a.message.t
				desc = giantess_feast.1000.a.message.desc
			}
		}
		ai_chance = {
			base = 10
			modifier = {
				faith = { has_doctrine_parameter = giantess_worshipped }
				add = 100
			}
			opinion_modifier = {
				opinion_target = scope:giantess
			}
		}
	}
	#Send prisoners
	option = {
		trigger = {
			any_prisoner = {
				age >= 18
				NOT = { has_trait = giantess }
				count >= 3
			}
		}
		name = giantess_feast.1000.b
		giantess_send_prisoners_feast_effect = yes
		tiny_loyalty_change = { CHANGE = minor_loyalty_gain }
		if = {
			limit = {
				faith = { has_doctrine_parameter = giantess_worshipped }
			}
			add_piety = medium_piety_gain
		}
		scope:giantess = {
			send_interface_message = {
				left_icon = scope:vassal
				type = event_prowess_good_desc
				title = giantess_feast.1000.b.message.t
				desc = giantess_feast.1000.b.message.desc
			}
		}
		ai_chance = {
			base = 10
			modifier = {
				faith = { has_doctrine_parameter = giantess_worshipped }
				add = 100
			}
			opinion_modifier = {
				opinion_target = scope:giantess
			}
		}
	}
	#Go yourself
	option = {
		trigger = {
			is_available_for_activity_trigger = yes
		}
		name = giantess_feast.1000.c
		add_character_flag = {
			flag = willing_vore
			days = 200
		}
		#imprison_character_effect = {
		#	TARGET = scope:vassal
		#	IMPRISONER = scope:giantess
		#	type = house_arrest
		#}
		scope:giantess = {
			imprison = { 
				target = scope:vassal
				type = house_arrest
			}
		}
		if = {
			limit = {
				faith = { has_doctrine_parameter = giantess_worshipped }
			}
			add_piety = major_piety_gain
		}
		scope:giantess = {
			send_interface_message = {
				left_icon = scope:vassal
				type = event_prowess_good_desc
				title = giantess_feast.1000.c.message.t
				desc = giantess_feast.1000.c.message.desc
			}
		}
		ai_chance = {
			base = 0
			modifier = {
				faith = { has_doctrine_parameter = giantess_worshipped }
				add = 20
			}
			modifier = {
				faith = { 
					AND = {
						has_doctrine_parameter = giantess_worshipped
						has_doctrine_parameter = tiny_toys
					}
				}
				add = 100
			}
			ai_value_modifier = {
				ai_boldness = 1
				min = 0
			}
			opinion_modifier = {
				opinion_target = scope:giantess
			}
		}
	}
	#Send nothing
	option = {
		name = giantess_feast.1000.d
		trigger = {
			NOT = { 
				has_opinion_modifier = {
					target = scope:giantess
					modifier = tiny_loyal
				}
			 }
		}
		if = {
			limit = {
				faith = {
					has_doctrine_parameter = giantess_worshipped
				}
			}
			add_piety = major_piety_loss
		}
		tiny_loyalty_change = { CHANGE = minor_loyalty_loss }
		scope:giantess = {
			add_opinion = {
				target = scope:vassal
				modifier = giantess_vassal_neglected_feast
			}
			send_interface_message = {
				left_icon = scope:vassal
				type = event_prowess_good_desc
				title = giantess_feast.1000.d.message.t
				desc = giantess_feast.1000.d.message.desc
			}
		}
		ai_chance = {
			base = 50
			ai_value_modifier = {
				ai_boldness = 0.5
				min = -50
			}
		}
	}
