					scope:torture_method = flag:boobsquish

				desc = giantess_torture.0002.desc.boobsquish.gentle
			}

				trigger = {
					scope:torture_method = flag:kiss

				desc = giantess_torture.0002.desc.kiss.gentle
			}

				trigger = {
					scope:torture_method = flag:kiss

				}
				desc = giantess_torture.0002.desc.vagina.gentle
			}
		}

	option = {
		name = giantess_torture.0002.desc.a
		show_as_tooltip = {
			scope:recipient = {
				stress_impact = {
					base = var:stress_change
				}
				random_list = {
					100 = {
						modifier = {
							add = {
								value = var:stress_change
								multiply = -1
							}
						}
					}
					0 = {
						trigger = {
							scope:actor.var:alignment = flag:evil
						}
						modifier = {
							add = var:stress_change
						}
						if = {
							limit = {
								has_trait = wounded
							}
							change_trait_rank = {
								trait = wounded
								rank = 1
								max = 3
							}
						}
						else = {
							add_trait = wounded_1
						}
					}
				}
			}
		}
		scope:recipient = {
			trigger_event = giantess_torture.0003
		}
	}
}

#Give up or not
giantess_torture.0004 = {
	type = character_event
	title = giantess_torture.0004.t
	left_portrait = scope:actor
	right_portrait = {
		character = scope:recipient
		animation = fear
	}
	theme = prison
	override_background = {
		reference = corridor_day
	}
	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					stress >= 300
				}
				desc = giantess_torture.0004.desc.surrender
			}
		}
		desc = giantess_torture.0004.desc
	}
	#Don't give up
	option = {
		name = giantess_torture.0004.a
		trigger = {
			stress < 300
		}
		ai_chance = {
			base = 100
			modifier = {
				stress >= 100
				add = -20
			}
			modifier = {
				stress >= 200
				add = -20
			}
		}
	}
	#Surrender
	option = {
		name = giantess_torture.0004.b
		make_loyal_effect = {
			TINY = scope:recipient
			GIANTESS = scope:actor
		}
		stress_impact = {
			base = major_stress_impact_loss
		}
		scope:actor = {
			send_interface_message = {
				left_icon = scope:recipient
				type = event_prowess_good_desc
				title = tiny_surrenders_prison.t
				desc = tiny_surrenders_prison.desc
			}
		}
		ai_chance = {
			base = 10
		}
		release_from_prison = yes
	}
}
