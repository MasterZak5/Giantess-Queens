				limit = {
					age >= 18

				}
				save_scope_as = tiny
			}
		}
		
	}
	option = {
		name = giantess_travel.0001.a
		trigger_event = {
			id = giantess_travel.0002
			days = { 1 3 }
		}
	}
}
