#Choose to reward or punish
giantess_reward_or_punish.0001 = {
	title = giantess_reward_or_punish.0001.t
	desc = giantess_reward_or_punish.0001.desc
	left_portrait = scope:actor
	right_portrait = scope:recipient
	#Reward
	option = {
		trigger = {
			scope:recipient.var:tiny_loyaty >= minor_loyalty_cost
		}
		name = giantess_reward_or_punish.0001.desc.a
		trigger_event = giantess_reward_or_punish.0002
	}
	#Punish
	option = {
		trigger = {
			scope:recipient.var:tiny_loyaty <= unloyalty_threshold
		}
		name = giantess_reward_or_punish.0001.desc.b
	}
	#Leave
	option = {
		name = giantess_reward_or_punish.0001.desc.c
	}
}

#Training, giantess
giantess_reward_or_punish.0005 = {
	title = giantess_reward_or_punish.0005.t
	desc = {
		desc = giantess_reward_or_punish.0005.desc.opening
		first_valid = {
			triggered_desc = {
				trigger = {
					scope:training_type = flag:ass_training
				}
				desc = giantess_reward_or_punish.0005.desc.ass
			}
			triggered_desc = {
				trigger = {
					scope:training_type = flag:vore_training
				}
				desc = giantess_reward_or_punish.0005.desc.vore
			}
			triggered_desc = {
				trigger = {
					scope:training_type = flag:boobs_training
				}
				desc = giantess_reward_or_punish.0005.desc.boobs
			}
			triggered_desc = {
				trigger = {
					scope:training_type = flag:foot_training
				}
				desc = giantess_reward_or_punish.0005.desc.foot
			}
			triggered_desc = {
				trigger = {
					scope:training_type = flag:vagina_training
				}
				desc = giantess_reward_or_punish.0005.desc.vagina
			}
		}
	}
	left_portrait = scope:actor
	right_portrait = scope:recipient
	option = {
		name = giantess_reward_or_punish.0005.desc.a
		scope:recipient = {
			trigger_event = giantess_reward_or_punish.0006
		}
		show_as_tooltip = {
			scope:recipient = {
				if = {
					limit = {
						scope:training_type = flag:ass_training
					}
					add_ass_xp = { CHANGE = major_xp_gain }
				}
				if = {
					limit = {
						scope:training_type = flag:vore_training
					}
					add_vore_xp = { CHANGE = major_xp_gain }
				}
				if = {
					limit = {
						scope:training_type = flag:boobs_training
					}
					add_boobs_xp = { CHANGE = major_xp_gain }
				}
				if = {
					limit = {
						scope:training_type = flag:foot_training
					}
					add_foot_xp = { CHANGE = major_xp_gain }
				}
				if = {
					limit = {
						scope:training_type = flag:vagina_training
					}
					add_vagina_xp = { CHANGE = major_xp_gain }
				}
				add_opinion = {
					target = scope:actor
					modifier = tiny_received_minor_reward
				}
			}
		}
	}
}
