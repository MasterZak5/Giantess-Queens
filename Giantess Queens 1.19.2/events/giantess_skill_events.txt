#Skills recount
giantess_skill.0012 = {
	hidden = yes
	trigger = {
		OR = {
			has_variable = skill_specialization 
			faith = { 
				religion_tag = allmother_worship_religion 
			}
		}
	}
	immediate = {
		if = {
			limit = {
				has_trait = giantess
			}
		}
		while = {
			limit = { var:ass_xp >= skill_levelup_threshold }
			change_variable = {
				name = ass_xp
				subtract = skill_levelup_threshold
			}
			add_ass_skill = { CHANGE = 1 }
		}
		while = {
			limit = { var:vore_xp >= skill_levelup_threshold }
			change_variable = {
				name = vore_xp
				subtract = skill_levelup_threshold
			}
			add_vore_skill = { CHANGE = 1 }
		}
		while = {
			limit = { var:foot_xp >= skill_levelup_threshold }
			change_variable = {
				name = foot_xp
				subtract = skill_levelup_threshold
			}
			add_foot_skill = { CHANGE = 1 }
		}
		while = {
			limit = { var:boobs_xp >= skill_levelup_threshold }
			change_variable = {
				name = boobs_xp
				subtract = skill_levelup_threshold
			}
			add_boobs_skill = { CHANGE = 1 }
		}
		while = {
			limit = { var:vagina_xp >= skill_levelup_threshold }
			change_variable = {
				name = vagina_xp
				subtract = skill_levelup_threshold
			}
			add_vagina_skill = { CHANGE = 1 }
		}
	}
	after = {
		ass_lifestyle_rank_up_check_effect = yes
		vore_lifestyle_rank_up_check_effect = yes
		feet_lifestyle_rank_up_check_effect = yes
	}
}
