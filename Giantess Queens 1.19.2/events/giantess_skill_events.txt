giantess_skill.0011 = {
	type = character_event
	theme = crown
	left_portrait = ROOT
	title = giantess_skill.0011.t
	desc = giantess_skill.0011.desc

	option = {
		name = giantess_skill.0011.desc.a

			name = alignment
			value = flag:gentle
