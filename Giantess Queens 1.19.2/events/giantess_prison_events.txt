# Escape from prison event chain

#Does your attempt succeed or not? Almost the same as prison.9000, except triggers additional event chain on success
giantess_prison.0001 = {
	hidden = yes

	immediate = {
		save_scope_as = escapee
		imprisoner = {
			save_scope_as = imprisoner
		}
		random_list = {
			50 = {
				trigger_event = giantess_prison.0002
				min = 5
				modifier = {
					add = {
						value = prowess
						subtract = average_skill_rating
						multiply = 2
					}
				}
				modifier = {
					add = 15
					has_character_modifier = childhood_prison_escape
				}
				modifier = {
					imprisoner.domicile ?= {
						is_domicile_type = camp
						NOT = { has_domicile_parameter = camp_reduced_prisoner_escape_chance }
					}
					factor = 2
				}
				ai_value_modifier = {
					ai_energy = 0.5
				}
			}
			50 = {
				trigger_event = {
					on_action = escape_from_prison_failure
				}
				min = 5
				modifier = {
					add = {
						value = scope:imprisoner.intrigue
						subtract = average_skill_rating
						multiply = 2
					}
				}
				modifier = {
					add = 25
					scope:imprisoner = {
						any_courtier = { suitable_shieldmaiden_gaoler_trigger = yes }
					}
				}
				modifier = {
					add = 25
					OR = {
						has_trait = weak
						has_trait = craven
						has_trait = physique_bad_1
					}
				}
				modifier = {
					add = 50
					OR = {
						has_trait = physique_bad_2
						has_trait = physique_bad_3
					}
				}
				modifier = {
					add = 25
					is_in_prison_type = dungeon
				}
			}
		}
	}
}

# Successful escape
giantess_prison.0003 = {
	type = character_event
	title = giantess_prison.0003.t
	desc = giantess_prison.0003.desc

	left_portrait = {
		character = root
		animation = ecstasy
	}
	lower_right_portrait = {
		character = scope:imprisoner
	}
	theme = prison
	override_background = {
		reference = wilderness_scope
	}


	immediate = {
		escape_from_prison_effect = yes
	}

	option = {
		name = giantess_prison.0003.a
	}
	after = {
		remove_character_flag = block_for_prison_release_notification
		if = { #To inform player imprisoner about the escape
			limit = {
				scope:imprisoner = { is_ai = no }
			}
			save_scope_as = escapee
			scope:imprisoner = { trigger_event = prison.9499 }
		}
	}
}

	}


	immediate = {

	option = {

			add_character_flag = {

	 	}

			limit = {
				scope:recipient = { is_ai = no } # Give players a chance to see what's about to happen
			}

				trigger_event = giantess_execution.0001 
			} 
		}
		else = { # Giantess receives event
			scope:actor = { 
				trigger_event = giantess_execution.0002 
				show_as_tooltip = {
					giantess_execute_prisoner_effect = {
						VICTIM = scope:recipient
						EXECUTIONER = scope:actor
