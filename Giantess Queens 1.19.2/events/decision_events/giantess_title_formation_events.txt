# Forming Titanes
##################################################
namespace = titanes_formation

# Hub event for choosing path.
titanes_formation.0001 = {
	type = character_event
	title = titanes_formation.0001.t
	desc = {
		desc = titanes_formation.0001.desc.intro
		first_valid = {
			# Doing it as the Greeks.
			triggered_desc = {
				trigger = {
					root.culture = culture:greek
				}
				desc = titanes_formation.0001.desc.greeks
			}
			# Doing it as somebody else.
			desc = titanes_formation.0001.desc.other
		}
	}
	theme = realm
	right_portrait = {
		character = root
		animation = personality_honorable
	}
	option = {
		name = titanes_formation.0001.a
        trigger_event = titanes_formation.0002
	}
    option = {
		name = titanes_formation.0001.b
        trigger_event = titanes_formation.0003
	}
    option = {
		name = titanes_formation.0001.c
	}
}
# Title and religion
titanes_formation.0002 = {
	type = character_event
	title = titanes_formation.0002.t
	desc = titanes_formation.0002.desc
	theme = realm
	right_portrait = {
		character = root
		animation = personality_bold
	}
	immediate = {
		play_music_cue = "mx_cue_epic_sacral_moment"
		create_titanes_scripted_effect = yes
        add_piety = 1000
        add_prestige = 1000
        add_legitimacy_effect = { LEGITIMACY = massive_legitimacy_gain}        
		set_character_faith = faith:greek_titans
	}
	option = {
		name = titanes_formation.0002.a
		set_nickname_effect = { NICKNAME = nick_the_titaness }
		# Send narrative fluff to other players (if any).
		every_player = {
			limit = {
				this != root
				is_within_diplo_range = { CHARACTER = root }
			}
            trigger_event = titanes_formation.0004
		}
	}
}
