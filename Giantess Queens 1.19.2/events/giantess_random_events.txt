#Pregnancy
giantess_random.0003 = {
	trigger = {
		is_married = yes
		is_pregnant = no
		any_consort = {
			is_male = yes
		}
	}
	title = giantess_random.0003.t
	desc = giantess_random.0003.desc
	theme = love
	left_portrait = root
	right_portrait = scope:tiny
	immediate = {
		random_consort = {
			limit = {
				NOT = { has_trait = giantess }
				is_male = yes
			}
			save_scope_as = tiny
		}
		save_scope_as = giantess
		scope:tiny = {
			if = {
				limit = {
					NOT = {
						has_variable = vagina_skill
					}
				}
				init_tiny_skills = yes
			}
		}
	}
	#Have sex
	option = {
		name = giantess_random.0003.desc.a
		random_list = {
			#Win
			60 = {
				modifier = {
					scope:giantess = { has_variable = vagina_skill }
					add = scope:giantess.var:vagina_skill
				}
				modifier = {
					scope:tiny = { has_variable = vagina_skill }
					add = scope:tiny.var:vagina_skill
				}
				save_scope_value_as = {
					name = sex_status
					value = flag:success
				}
			}
			#Fail
			100 = {
				modifier = {
					scope:giantess = { has_variable = vagina_skill }
					add = {
						value = scope:giantess.var:vagina_skill
						multiply = -1
					}
				}
				modifier = {
					scope:tiny = { has_variable = vagina_skill }
					add = {
						value = scope:tiny.var:vagina_skill
						multiply = -1
					}
				}
				save_scope_value_as = {
					name = sex_status
					value = flag:fail
				}
			}
		}
		trigger_event = giantess_random.0004
		scope:tiny = {
			trigger_event = giantess_random.0005
		}
		ai_chance = {
			base = 100
		}
	}
	#No
	option = {
		name = giantess_random.0003.desc.b
		ai_chance = {
			base = 0
		}
	}
}
