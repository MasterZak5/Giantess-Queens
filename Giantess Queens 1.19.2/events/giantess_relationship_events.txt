namespace = giantess_relationship

# Horny Giantess - 1
giantess_relationship.0001 = {
    type = character_event
	title = giantess_relationship.0001.t
	desc = giantess_relationship.0001.desc
	theme = family
    #cooldown = { years = 3 }
    override_background = { reference = throne_room }
    
	trigger = {
	    is_ai = no
        is_adult = yes
        has_trait = giantess
        any_spouse ?= {
            is_physically_able_ai_adult = yes
            NOT = { has_trait = giantess }
        }
	}
	left_portrait = {
        character = root
        animation = boredom
        camera = camera_event_scheme_vs_right_look_right
    }
    lower_right_portrait = {
        character = scope:SPOUSE
    }
    immediate = {
        if = {
            limit = {
                any_spouse = {
                    NOT = { has_trait = giantess }
                    is_physically_able_ai_adult = yes
                }
            }
            random_spouse = {
                limit = {
                    NOT = { has_trait = giantess }
                    is_physically_able_ai_adult = yes
                }
                save_scope_as = SPOUSE
            }
        }
    }

    option = {
        name = giantess_relationship.0001.a
        trigger_event = { id = giantess_relationship.0002 }
    }
}
