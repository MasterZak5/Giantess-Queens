#Choose execution type
giantess_execution.0003 = {
	type = character_event
	title = giantess_execution.0003.t
	theme = prison
	override_background = {
		reference = gallows
	}
	left_portrait = {
		character = scope:actor
		animation = dismissal
	}
	right_portrait = {
		character = scope:recipient
		animation = beg
	}
	desc = giantess_execution.0003_desc

	#Ass
	option = {
		name = giantess_execution.0003.a
		scope:recipient = { 
			add_character_flag = {
				flag = execution_ass
				days = 1
			}
	 	}
		if = {
			limit = {
				scope:recipient = { is_ai = no } # Give players a chance to see what's about to happen
			}
			scope:recipient = { 
				trigger_event = giantess_execution.0001 
			} 
		}
		else = { # Giantess receives event
			scope:actor = { 
				trigger_event = giantess_execution.0002 
				show_as_tooltip = {
					giantess_execute_prisoner_effect = {
						VICTIM = scope:recipient
						EXECUTIONER = scope:actor
					}
				}
			}
		}
		ai_chance = {
			base = 100
		}
	}
	#Swallow
	option = {
		name = giantess_execution.0003.b
		scope:recipient = { 
			add_character_flag = {
				flag = execution_swallow
				days = 1
			}
	 	}
		if = {
			limit = {
				scope:recipient = { is_ai = no } # Give players a chance to see what's about to happen
			}
			scope:recipient = { 
				trigger_event = giantess_execution.0001 
			} 
		}
		else = { # Giantess receives event
			scope:actor = { 
				trigger_event = giantess_execution.0002 
				show_as_tooltip = {
					giantess_execute_prisoner_effect = {
						VICTIM = scope:recipient
						EXECUTIONER = scope:actor
					}
				}
			}
		}
		ai_chance = {
			base = 100
		}
	}
	#Shoecrush
	option = {
		name = giantess_execution.0003.c
		scope:recipient = { 
			add_character_flag = {
				flag = execution_shoecrush
				days = 1
			}
	 	}
		if = {
			limit = {
				scope:recipient = { is_ai = no } # Give players a chance to see what's about to happen
			}
			scope:recipient = { 
				trigger_event = giantess_execution.0001 
			} 
		}
		else = { # Giantess receives event
			scope:actor = { 
				trigger_event = giantess_execution.0002 
				show_as_tooltip = {
					giantess_execute_prisoner_effect = {
						VICTIM = scope:recipient
						EXECUTIONER = scope:actor
					}
				}
			}
		}
		ai_chance = {
			base = 100
		}
	}
	#Unbirth
	option = {
		name = giantess_execution.0003.d
		scope:recipient = {
			add_character_flag = {
				flag = execution_unbirth
				days = 1
			}
		}
		trigger = {
			scope:recipient = { is_male = yes }
			NOT = { scope:actor = {	has_character_modifier = recently_unbirthed } }
			NOT = { scope:actor = {	is_pregnant = yes } }
		}
		scope:recipient = {
			add_character_flag = {
				flag = execution_unbirth
				days = 1
			}
	 	}
		if = {
			limit = {
				scope:recipient = { is_ai = no } # Give players a chance to see what's about to happen
			}
			scope:recipient = {
				trigger_event = giantess_execution.0001
			}
		}

			scope:actor = {
				trigger_event = giantess_execution.0002
				show_as_tooltip = {
					giantess_execute_prisoner_effect = {
						VICTIM = scope:recipient
						EXECUTIONER = scope:actor
					}
				}
			}
		}
		ai_chance = {
			base = 100
		}
	}
}


# Executed! (Player by AI)
giantess_execution.0001 = {
	type = character_event
	title = giantess_execution.0001.t
	theme = prison
	override_background = {
		reference = gallows
	}

	desc = {
		first_valid = {
			triggered_desc = {
				trigger = {
					scope:recipient = { has_character_flag = execution_ass }
				}
				desc = giantess_execution.0001.desc_ass
			}
			triggered_desc = {
				trigger = {
					scope:recipient = { has_character_flag = execution_swallow }
				}
				desc = giantess_execution.0001.desc_swallow
			}
			triggered_desc = {
				trigger = {
					scope:recipient = { has_character_flag = execution_shoecrush }
				}
				desc = giantess_execution.0001.desc_shoecrush
			}
			triggered_desc = {
				trigger = {
					scope:recipient = { has_character_flag = execution_unbirth }
				}
				desc = giantess_execution.0001.desc_unbirth
			}
		}
	}
	
	left_portrait = {
		character = scope:actor
		animation = dismissal
	}

	right_portrait = {
		character = scope:recipient
		animation = beg
	}
	immediate = {
		play_music_cue = mx_cue_stress
	}

	option = {
		name = giantess_execution.0001.a

		giantess_execute_prisoner_effect = {
			VICTIM = scope:recipient
			EXECUTIONER = scope:actor
		}
	}
}
